KMeans¶

In [19]:
x=iris_df.iloc[:,[0,1,2,3]].values
sse = []
k_rng = range(1,10)
for k in k_rng:
    km = KMeans(n_clusters=k)
    km.fit(x)
    sse.append(km.inertia_)
